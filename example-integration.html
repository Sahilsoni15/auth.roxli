<!-- Include Roxli SDK -->
<script src="http://localhost:5000/static/js/roxli-sdk.js"></script>

<!-- Add login button -->
<button onclick="signInWithRoxli()" class="roxli-signin-btn">
    Sign in with Roxli
</button>

<script>
function signInWithRoxli() {
    RoxliAuth.signIn({
        onSuccess: function(user) {
            showUserInfo(user);
            showStatus('Login successful!', 'success');
        },
        onError: function(error) {
            showStatus('Login failed: ' + error, 'error');
        }
    });
}

function signOut() {
    RoxliAuth.signOut().then(() => {
        showAuthSection();
        showStatus('Signed out successfully', 'success');
    });
}
</script>

üîç Check Authentication Status:

// Check if user is already logged in
const user = RoxliAuth.getUser();
if (user) {
    console.log('User is logged in:', user);
    showUserInfo(user);
} else {
    console.log('User is not logged in');
    showAuthSection();
}

// Listen for authentication changes
RoxliAuth.onAuthStateChanged(function(user) {
    if (user) {
        showUserInfo(user);
    } else {
        showAuthSection();
    }
});
